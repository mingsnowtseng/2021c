##基本加減, 總格數(%格數d)
```C
#include <stdio.h>
int main()
{
    int a=123,b=10;
    printf("%d\n",a+b);
    printf("%5d\n",a+b);
}

##浮點數 (%f)
```C
#include <stdio.h>
int main()
{
    ///int a,b;
    float  a=3.14159;
    double b=3.14159;
    printf("%f\n", a);
    printf("%.2f\n", a);
}

##成績判斷 (if)
```C
#include <stdio.h>
int main()
{
	int n;
	scanf("%d",&n);
	if(n>=90) printf("A+");
	else if(n>=80) printf("A");
	else if(n>=70) printf("B");
	else if(n>=60) printf("C");
	else printf("D");
}
